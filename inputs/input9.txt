fun main() {
    val byteVal: Byte = 10
    val shortVal: Short = 20
    val intVal: Int = 30
    val longVal: Long = 40L
    
    val sumInt: Int = byteVal + shortVal + intVal
    println(sumInt)
    
    val sumLong: Long = byteVal.toLong() + shortVal.toLong() + intVal.toLong() + longVal
    println(sumLong)
    
    val floatVal: Float = 1.23F
    val doubleVal: Double = 4.56
    
    val sumDouble: Double = floatVal.toDouble() + doubleVal
    println(sumDouble)
    
    val doubleToInt: Int = doubleVal.toInt()
    println(doubleToInt)
}
